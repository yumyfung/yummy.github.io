gulp-next

gulp-next插件用于继续在管道里执行任务。

gulp-next github托管

安装：

npm install --save-dev gulp-next

参数：

callback：回调函数

使用：

					var gulp = require('gulp');
					var next= require('gulp-next');

					gulp.task('test', function () {
					  gulp.src('images/*.png')
					      .pipe(gulp.dest('path/output/'))
					      .pipe(next(function(){
					          console.log('success...');
					      }));
					});
